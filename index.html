Guía de configuración 5 minutos · una sola vez
1
Abrir Google Apps Script
Ve a script.google.com → "Nuevo proyecto"
2
Pegar el código del script v3
Borra todo (Ctrl+A) y pega este código actualizado:
// TYC Servicios — Drive Sync v3 (2026) const FOLDER_ID = '1Z7HuRjfjWXOgjwBX9GElgXHFnqV2B9pm'; const FILE_NAME = 'TYC_Materiales_2026.json'; const LOG_NAME = 'TYC_SyncLog_2026.json'; function doGet(e) { try { const list = loadFromFile(FILE_NAME); return buildResponse({ok:true,productos:list,count:list.length,timestamp:new Date().toISOString()}); } catch(err) { return buildResponse({ok:false,msg:err.message}); } } function doPost(e) { try { const body = JSON.parse(e.postData.contents); const action = body.action; if (action === 'sync_all') { if (!body.productos || !Array.isArray(body.productos)) return buildResponse({ok:false,msg:'productos no es un array'}); saveToFile(body.productos, FILE_NAME); appendLog('sync_all',{count:body.productos.length}); return buildResponse({ok:true,count:body.productos.length,timestamp:new Date().toISOString()}); } if (action === 'add_product') { const list = loadFromFile(FILE_NAME); const p = body.producto; if (!p || !p.name) return buildResponse({ok:false,msg:'producto inválido'}); if (!p.n) p.n = list.length + 1; list.push(p); saveToFile(list, FILE_NAME); appendLog('add_product',{n:p.n,name:p.name}); return buildResponse({ok:true,total:list.length}); } if (action === 'update_product') { const list = loadFromFile(FILE_NAME); const p = body.producto; if (!p || !p.n) return buildResponse({ok:false,msg:'falta producto.n'}); const idx = list.findIndex(x => x.n === p.n); if (idx >= 0) list[idx] = p; else list.push(p); saveToFile(list, FILE_NAME); appendLog('update_product',{n:p.n,name:p.name,price_sin:p.price_sin}); return buildResponse({ok:true}); } if (action === 'delete_product') { let list = loadFromFile(FILE_NAME); const found = list.find(x => x.n === body.n); if (!found) return buildResponse({ok:false,msg:'Producto no encontrado'}); list = list.filter(x => x.n !== body.n).map((x,i) => ({...x,n:i+1})); saveToFile(list, FILE_NAME); appendLog('delete_product',{name:found.name}); return buildResponse({ok:true,total:list.length}); } return buildResponse({ok:false,msg:'Acción no reconocida: '+action}); } catch(err) { return buildResponse({ok:false,msg:err.message}); } } function buildResponse(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function saveToFile(data, name) { const folder = DriveApp.getFolderById(FOLDER_ID); const files = folder.getFilesByName(name); const content = JSON.stringify(data, null, 2); if (files.hasNext()) files.next().setContent(content); else folder.createFile(name, content, MimeType.PLAIN_TEXT); } function loadFromFile(name) { try { const folder = DriveApp.getFolderById(FOLDER_ID); const files = folder.getFilesByName(name); if (!files.hasNext()) return []; const txt = files.next().getBlob().getDataAsString('UTF-8'); if (!txt || txt.trim() === '') return []; return JSON.parse(txt); } catch(e) { return []; } } function appendLog(action, details) { try { const log = loadFromFile(LOG_NAME); log.unshift({action,details,timestamp:new Date().toISOString(), fecha:Utilities.formatDate(new Date(),'America/Lima','dd/MM/yyyy HH:mm')}); if(log.length > 200) log.splice(200); saveToFile(log, LOG_NAME); } catch(e){} } // Ejecuta esto en el editor para probar antes de publicar: function testConexion() { const folder = DriveApp.getFolderById(FOLDER_ID); const list = loadFromFile(FILE_NAME); Logger.log('✓ Carpeta: ' + folder.getName()); Logger.log('✓ Productos: ' + list.length); }
